CREATE TABLE emp_role (
    emp_role_id INT NOT NULL,
    emp_role_name VARCHAR(50),
    salary float(10, 2),
    PRIMARY KEY (emp_role_id)
);

CREATE TABLE password_status (
    password_status_id VARCHAR(1) NOT NULL,
    password_status_name VARCHAR(50),
    PRIMARY KEY (password_status_id)
);

CREATE TABLE emp_status (
    emp_status_id VARCHAR(1) NOT NULL,
    emp_status_name VARCHAR(50),
    PRIMARY KEY (emp_status_id)
);

CREATE TABLE table_status (
    table_status_id VARCHAR(1) NOT NULL,
    table_status_name VARCHAR(50),
    PRIMARY KEY (table_status_id)
);

CREATE TABLE order_status (
    order_status_id VARCHAR(1) NOT NULL,
    order_status_name VARCHAR(50),
    PRIMARY KEY (order_status_id)
);

CREATE TABLE order_food_status (
    serve_status_id VARCHAR(1) NOT NULL,
    serve_status_name VARCHAR(50),
    PRIMARY KEY (serve_status_id)
);

CREATE TABLE status_advice (
    status_advice_id VARCHAR(1) NOT NULL,
    status_advice_name VARCHAR(50),
    PRIMARY KEY (status_advice_id)
);

CREATE TABLE borrow_status (
    borrow_status_id VARCHAR(1) NOT NULL,
    borrow_status_name VARCHAR(50),
    PRIMARY KEY (borrow_status_id)
);

CREATE TABLE status_ingredient (
    status_ingredient_id VARCHAR(1) NOT NULL,
    status_ingredient_name VARCHAR(50),
    PRIMARY KEY (status_ingredient_id)
);

CREATE TABLE catagory_board_game (
    catagory_bg_id INT NOT NULL,
    catagory_bg_name VARCHAR(255),
    PRIMARY KEY (catagory_bg_id)
);

CREATE TABLE board_game_play (
    bgp_id INT NOT NULL,
    bgp_name VARCHAR(255),
    quantity INT,
    img_game_play VARCHAR(255),
    PRIMARY KEY (bgp_id)
);

CREATE TABLE board_game_sale (
    bg_id INT NOT NULL,
    bg_name VARCHAR(255),
    quantity INT,
    price DECIMAL(10, 2),
    img_game_sale VARCHAR(255),
    PRIMARY KEY (bg_id)
);


CREATE TABLE food_type_ (
    food_type_id INT NOT NULL,
    food_type_name VARCHAR(255),
    PRIMARY KEY (food_type_id)
);

CREATE TABLE food_list (
    food_id INT NOT NULL,
    food_name VARCHAR(255),
    food_type_id INT, -- ฟิลด์ที่เพิ่มเข้ามาใหม่ตามรูปภาพ
    PRIMARY KEY (food_id),
    FOREIGN KEY (food_type_id) REFERENCES food_type_(food_type_id)
);

CREATE TABLE variants (
    variant_id INT NOT NULL,
    variant_name VARCHAR(255),
    PRIMARY KEY (variant_id)
);

CREATE TABLE food_options (
    options_id INT NOT NULL,
    option_name VARCHAR(255),
    option_price DECIMAL(10, 2),
    PRIMARY KEY (options_id)
);

CREATE TABLE employee (
    emp_id INT NOT NULL,
    password VARCHAR(255),
    emp_first_name VARCHAR(255),
    emp_last_name VARCHAR(255),
    emp_role_id INT,
    age INT,
    sex VARCHAR(10),
    emp_status_id VARCHAR(1),
    password_status_id VARCHAR(1),
    PRIMARY KEY (emp_id),
    FOREIGN KEY (emp_role_id) REFERENCES emp_role(emp_role_id),
    FOREIGN KEY (emp_status_id) REFERENCES emp_status(emp_status_id),
    FOREIGN KEY (password_status_id) REFERENCES password_status(password_status_id)
);

CREATE TABLE table_no (
    table_number INT NOT NULL,
    table_status_id VARCHAR(1),
    PRIMARY KEY (table_number),
    FOREIGN KEY (table_status_id) REFERENCES table_status(table_status_id)
);

CREATE TABLE ingredient (
    ingredient_id INT NOT NULL,
    ingredient_name VARCHAR(255),
    status_ingredient_id VARCHAR(1),
    PRIMARY KEY (ingredient_id),
    FOREIGN KEY (status_ingredient_id) REFERENCES status_ingredient(status_ingredient_id)
);

CREATE TABLE food_variants (
    food_variant_id INT NOT NULL,
    food_id INT,
    food_variant_price DECIMAL(10, 2),
    img_food_url VARCHAR(255),
    price DECIMAL(10, 2),
    variant_id INT,
    PRIMARY KEY (food_variant_id),
    FOREIGN KEY (food_id) REFERENCES food_list(food_id),
    FOREIGN KEY (variant_id) REFERENCES variants(variant_id)
);

CREATE TABLE catagory_tag_id (
    category_tag_id INT NOT NULL,
    catagory_bg_id INT,
    bg_id INT,
    bgp_id INT,
    PRIMARY KEY (category_tag_id),
    FOREIGN KEY (catagory_bg_id) REFERENCES catagory_board_game(catagory_bg_id),
    FOREIGN KEY (bg_id) REFERENCES board_game_sale(bg_id),
    FOREIGN KEY (bgp_id) REFERENCES board_game_play(bgp_id)
);

CREATE TABLE `order` (
    order_id INT NOT NULL,
    table_number INT,
    date_time DATETIME,
    order_status_id VARCHAR(1),
    total_price DECIMAL(10, 2),
    emp_id INT,
    PRIMARY KEY (order_id),
    FOREIGN KEY (table_number) REFERENCES table_no(table_number),
    FOREIGN KEY (order_status_id) REFERENCES order_status(order_status_id),
    FOREIGN KEY (emp_id) REFERENCES employee(emp_id)
);

CREATE TABLE advice_list (
    advice_id INT NOT NULL,
    tablenumber INT,
    status_advice_id VARCHAR(1),
    date_time DATETIME,
    PRIMARY KEY (advice_id),
    FOREIGN KEY (tablenumber) REFERENCES table_no(table_number),
    FOREIGN KEY (status_advice_id) REFERENCES status_advice(status_advice_id)
);

CREATE TABLE borrow (
    borrow_id INT NOT NULL,
    table_number INT,
    bgp_id INT,
    date_time DATETIME,
    borrow_status_id VARCHAR(1),
    PRIMARY KEY (borrow_id),
    FOREIGN KEY (table_number) REFERENCES table_no(table_number),
    FOREIGN KEY (bgp_id) REFERENCES board_game_play(bgp_id),
    FOREIGN KEY (borrow_status_id) REFERENCES borrow_status(borrow_status_id)
);

CREATE TABLE bill_game (
    sale_bg_id INT NOT NULL,
    emp_id INT,
    date_time DATETIME,
    total_price DECIMAL(10, 2),
    PRIMARY KEY (sale_bg_id),
    FOREIGN KEY (emp_id) REFERENCES employee(emp_id)
);

CREATE TABLE order_food (
    order_detail_id INT NOT NULL,
    order_id INT,
    base_price DECIMAL(10, 2),
    quantity INT,
    food_variant_id INT,
    serve_status_id VARCHAR(1),
    PRIMARY KEY (order_detail_id),
    FOREIGN KEY (order_id) REFERENCES `order`(order_id),
    FOREIGN KEY (food_variant_id) REFERENCES food_variants(food_variant_id),
    FOREIGN KEY (serve_status_id) REFERENCES order_food_status(serve_status_id)
);

CREATE TABLE food_menu_options (
    food_variant_id INT NOT NULL,
    options_id INT NOT NULL,
    PRIMARY KEY (food_variant_id, options_id),
    FOREIGN KEY (food_variant_id) REFERENCES food_variants(food_variant_id),
    FOREIGN KEY (options_id) REFERENCES food_options(options_id)
);

CREATE TABLE order_food_options (
    order_detail_id INT NOT NULL,
    options_id INT NOT NULL,
    PRIMARY KEY (order_detail_id, options_id),
    FOREIGN KEY (order_detail_id) REFERENCES order_food(order_detail_id),
    FOREIGN KEY (options_id) REFERENCES food_options(options_id)
);

CREATE TABLE bill_game_detail (
    sale_bg_detail_id INT NOT NULL,
    sale_bg_id INT,
    bg_id INT,
    unit_price DECIMAL(10, 2),
    total_price DECIMAL(10, 2),
    quantity INT,
    PRIMARY KEY (sale_bg_detail_id),
    FOREIGN KEY (sale_bg_id) REFERENCES bill_game(sale_bg_id),
    FOREIGN KEY (bg_id) REFERENCES board_game_sale(bg_id)
);
